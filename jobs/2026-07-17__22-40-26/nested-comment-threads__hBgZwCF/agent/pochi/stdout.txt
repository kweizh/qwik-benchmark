import re
html = open('/tmp/page3.html').read()

# Find the start and end of each comment div
# Each comment has data-comment-id, so let's find balanced positions by tracking div tags
# Simpler approach: find each comment-id attribute position, and the closing </div> that
# closes the parent .comment div

# Find all comment-id positions
positions = [(int(m.group(1)), m.start()) for m in re.finditer(r'data-comment-id="(\d+)"', html)]
positions.sort(key=lambda x: x[1])

# For each comment, find its opening div tag position (the <div that has its data-comment-id)
# and find the closing </div> that matches it
def find_comment_end(comment_id):
    # Find the start of <div that has this comment-id
    idx = html.find(f'data-comment-id="{comment_id}"')
    # Walk backwards to find the opening <div
    div_start = html.rfind('<div', 0, idx)
    # Walk forward to count balanced divs
    pos = div_start
    depth = 0
    while pos < len(html):
        open_match = re.search(r'<div[\s>]', html[pos:])
        close_match = re.search(r'</div>', html[pos:])
        if not open_match and not close_match:
            break
        if open_match and (not close_match or open_match.start() < close_match.start()):
            depth += 1
            pos += open_match.end()
        else:
            depth -= 1
            pos += close_match.end()
            if depth == 0:
                return pos
    return -1

ends = {cid: find_comment_end(cid) for cid, _ in positions}
starts = {cid: p for cid, p in positions}

# Print whether each comment's start and end are properly nested
for cid in [1, 2, 3, 4]:
